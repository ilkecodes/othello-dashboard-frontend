// Step 4'te (Kampanya Detayı) client_id yerine dropdown ekle

import { useEffect } from 'react';
import { getClients } from '@/lib/api';

// Component içinde:
const [clients, setClients] = useState([]);

useEffect(() => {
  getClients().then(res => setClients(res.data)).catch(console.error);
}, []);

// Input yerine Select:
<div>
  <label className="text-sm font-medium">Müşteri Seçin</label>
  <Select value={formData.client_id} onValueChange={(v) => setFormData({...formData, client_id: v})}>
    <SelectTrigger>
      <SelectValue placeholder="Müşteri seçin" />
    </SelectTrigger>
    <SelectContent>
      {clients.map(client => (
        <SelectItem key={client.id} value={client.id}>
          {client.name} ({client.brand_guidelines?.industry})
        </SelectItem>
      ))}
    </SelectContent>
  </Select>
</div>
